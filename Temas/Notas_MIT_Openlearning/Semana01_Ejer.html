2.2) Transpuesta¶

def tp(A):
    return np.transpose(A)

def tp1(A):
    return (np.transpose(A),A.T)

a, b = tp1(A)
a
b

array([[ 1. , -1.8],
       [ 2.4,  9. ],
       [ 5. ,  2. ]])

    return tp(rv(value_list))

2.6)¶

def length(col_v):
    v2 = col_v * col_v
    v2_s = np.sum(v2)
    norm = v2_s ** 0.5
    return norm

length(cv([3,4]))

5.0

2.7)¶

def normalize(col_v):
    norm = length(col_v)
    col_v_u = (1/norm) * col_v 
    return col_v_u

normalize(cv([3,4]))

array([[0.6],
       [0.8]])

a = normalize(cv([3,4]))

length(a)

1.0
